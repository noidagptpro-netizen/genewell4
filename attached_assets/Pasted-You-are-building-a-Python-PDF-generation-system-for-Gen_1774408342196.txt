def calculate_bmr(weight_kg, height_cm, age, gender) -> int:
    """Mifflin-St Jeor. gender = 'male' or 'female'"""
    # Female: (10×w) + (6.25×h) - (5×age) - 161
    # Male:   (10×w) + (6.25×h) - (5×age) + 5

def calculate_tdee(bmr: int, activity_level: str) -> int:
    """
    activity_level options and multipliers:
    'sedentary'       → 1.2
    'lightly-active'  → 1.375
    'moderately-active' → 1.55
    'very-active'     → 1.725
    'shift-work'      → 1.375  (same as lightly active)
    """

def calculate_calorie_targets(tdee: int, goal: str) -> dict:
    """
    goal options:
    'gain-weight'    → surplus 250–400 kcal
    'lose-weight'    → deficit 300–500 kcal
    'maintain'       → tdee ± 50 kcal
    'lean-muscle'    → surplus 200–300 kcal
    Returns: {low, high, surplus_deficit, protein_g, carbs_g, fats_g}
    Macros: Protein=25%, Carbs=50%, Fats=25% of target calories
    1g protein=4kcal, 1g carb=4kcal, 1g fat=9kcal
    """

def calculate_water_target(weight_kg: float, activity: str) -> float:
    """
    Base: weight × 0.033 litres
    Add 0.3L if active/shift-work
    Returns float rounded to 1 decimal
    """

def calculate_health_scores(user_data: dict) -> dict:
    """
    Compute 0–100 scores for each dimension:
    sleep_score:    (sleep_hours / 8) × 100, capped at 100
    hydration_score:(water_L / water_target) × 100, capped at 100
    meal_score:     (meals_per_day / 3) × 100, capped at 100
    stress_score:   'low'→85, 'moderate'→50, 'high'→25
    gut_score:      base 80, -20 if constipation, -15 if bloating_frequent
    skin_score:     base 80, -20 if dry_skin, -10 if low hydration
    overall_score:  average of all 6 scores
    Returns dict with all scores + overall + status labels
    """

def get_score_status(score: int) -> str:

QUALITY REQUIREMENTS (non-negotiable)
═══════════════════════════════════════════════════════════════

1. COMPUTED, NOT HARDCODED
   Every number in the PDF must be derived from user_data inputs.
   Never hardcode values like "65 kg" or "2,139 kcal". 
   Always compute: calculate_bmr(user.weight, user.height, user.age, user.gender)

2. CONDITIONAL CONTENT
   Every section must show/hide or change content based on user flags:
   - Lactose intolerant → no dairy in meal plan, swap to coconut/almond
   - Thyroid condition → add thyroid-specific notes to sleep + supplement sections
   - Anxiety present → add breathwork + ashwagandha to relevant sections
   - DNA uploaded → add DNA section, else show "Upload DNA to unlock" callout

3. CHART QUALITY
   All matplotlib charts must use the PAGE_BG background.
   All text in charts must use NAVY color.
   All primary data must use GOLD color.
   Charts embedded as BytesIO PNG at 150 dpi.
   No chart titles should be cut off (use tight_layout).

4. TABLE QUALITY
   All tables must use card_table() from base.py.
   Header: NAVY background, WHITE text, Helvetica-Bold 9pt.
   Rows: CREAM / CREAM_DARK alternating.
   Grid: 0.3pt #CCCCCC lines.
   Padding: 6pt top/bottom, 8pt left/right.

5. PAGE FOOTER
   Every page must have the navy footer stripe via on_page().
   Left: GeneWell · genewell.in · educational purposes only
   Center: page number in gold
   Right: user name · order ID

6. FILE NAMING
   Output files named: {product_id}_{order_id}_{name_slug}.pdf
   Example: brainy-blueprint_abi711y7v_brainy.pdf

7. NO UNICODE SUBSCRIPTS
   Never use ₀₁₂₃₄₅₆₇₈₉ or ⁰¹²³ in reportlab.
   Use <sub> and <super> XML tags in Paragraph objects instead.

8. INDIAN CONTEXT
   All food items must be Indian names (dal, roti, sabzi, etc.)
   All prices in ₹ (use Rs. in reportlab if ₹ renders as box)
   All lab names must be Indian (Thyrocare, Dr. Lal PathLabs, Metropolis)
   All supplement brands must be available in India
